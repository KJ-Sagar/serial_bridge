#ifndef PICO_ESPAT_UDP_UTILS_H
#define PICO_ESPAT_UDP_UTILS_H

    #include <stdio.h>
    #include <string.h>
    #include <stdlib.h>
    #include "pico/stdlib.h"
    #include "hardware/uart.h"

    #define DEBUG_ENABLE

    #define UART_ID uart0
    #define UART_TX_PIN 0
    #define UART_RX_PIN 1
    #define BAUD_RATE 115200

    #define PER_POLL_PERIOD 100

    #define BUFFER_LENGTH 256
    char rx_buffer[BUFFER_LENGTH];

    void read_at_response() {
        int index = 0;  //position of string indicator
        while (uart_is_readable(UART_ID) && index < BUFFER_LENGTH - 1) {
            char c = uart_getc(UART_ID);
            // Completely ignore '\r'
            if (c == '\r') {
                continue;
            }

            // Ignore '\n' if it is the first character
            if (c == '\n') {
                if (index==0) {
                    continue;
                }
                else {
                    break;
                }
            }
            rx_buffer[index++] = c;
        }
        rx_buffer[index] = '\0';
    }

    #define RX_FIFO_SIZE 512
    static char uart_rx_fifo[RX_FIFO_SIZE];
    static int uart_rx_len = 0;

    bool read_ipd_stream(char* ipd_buffer, size_t ipd_len) {
        // Fetch all data from UART Buffer
        while (uart_is_readable(UART_ID) && uart_rx_len < RX_FIFO_SIZE) {
            uart_rx_fifo[uart_rx_len++] = uart_getc(UART_ID);
        }

        // Look for +IPD prefix
        for (int i = 0; i <= uart_rx_len - 5; i++) {
            if (strncmp(&uart_rx_fifo[i], "+IPD,", 5) == 0) {
                int j = i + 5;
                // parse connection id
                int conn_id = uart_rx_fifo[j] - '0';
                j++;

                if (uart_rx_fifo[j] != ',') return false;
                j++;

                // parse packet length
                int packet_len = 0;
                while (uart_rx_fifo[j] >= '0' && uart_rx_fifo[j] <= '9') {
                    packet_len = packet_len * 10 + (uart_rx_fifo[j] - '0');
                    j++;
                }

                if (uart_rx_fifo[j] != ':') return false;
                j++;

                // check if full packet is received
                if (j + packet_len > uart_rx_len) return false;

                // copy packet
                size_t copy_len = (packet_len >= ipd_len) ? ipd_len - 1 : packet_len;
                memcpy(ipd_buffer, &uart_rx_fifo[j], copy_len);
                ipd_buffer[copy_len] = '\0';

                // remove processed bytes
                int remaining = uart_rx_len - (j + packet_len);
                memmove(uart_rx_fifo, &uart_rx_fifo[j + packet_len], remaining);
                uart_rx_len = remaining;

                return true;
            }
        }

        return false;
    }

    void clear_uart_buffer() {
        while (uart_is_readable(UART_ID)) {
            uart_getc(UART_ID);
        }
    }

    bool at_send_command(char* command, int timeout) {
        // Send command
        clear_uart_buffer();
        uart_puts(uart0, command);

        // Check command status
        int n_retries = timeout / PER_POLL_PERIOD;
        for (int i=0; i<n_retries; i++) {
            read_at_response();
            if (strcmp(rx_buffer, "OK") == 0) {
                return true;
            }
            else if (strcmp(rx_buffer, "ERROR") == 0) {
                return false;
            }
            else {
#ifdef DEBUG_ENABLE
                if (strlen(rx_buffer)) printf("%s\n", rx_buffer);
#endif
            }
            sleep_ms(PER_POLL_PERIOD);
        }

        // If no response before timeout
        return false;
    }

    bool espat_init() {
        // Setup UART
        uart_init(UART_ID, BAUD_RATE);
        gpio_set_function(UART_TX_PIN, GPIO_FUNC_UART);
        gpio_set_function(UART_RX_PIN, GPIO_FUNC_UART);

        // Health check
        if (!at_send_command("AT\r\n", 1000)) return false;

        // Disable echo
        if (!at_send_command("ATE0\r\n", 1000)) return false;

        return true;
    }

    bool connect_wifi(char* wifi_ssid, char* wifi_password) {
        // Setup as station
        if (!at_send_command("AT+CWMODE=1\r\n", 1000)) return false;

        // Connect to WiFi
        char command[100];
        sprintf(command, "AT+CWJAP=\"%s\",\"%s\"\r\n", wifi_ssid, wifi_password);
        if (!at_send_command(command, 20000)) return false;

        return true;
    }

    bool disconnect_wifi() {
        if (!at_send_command("AT+CWQAP\r\n", 20000)) return false;

        return true;
    }

    bool start_tcp_server(int port) {
        // Allow multiple connections
        if (!at_send_command("AT+CIPMUX=1\r\n", 1000)) return false;

        // Start TCP server
        char command[100];
        sprintf(command, "AT+CIPSERVER=1,%d\r\n", port);
        if (!at_send_command(command, 1000)) return false;

        return true;
    }

    bool stop_tcp_server() {
        // Stop TCP server and close all connections
        if (!at_send_command("AT+CIPSERVER=0,1\r\n", 1000)) return false;
        
        // Disable multiple connections
        if (!at_send_command("AT+CIPMUX=0\r\n", 1000)) return false;

        return true;
    }


#endif //PICO_ESPAT_UDP_UTILS_H
